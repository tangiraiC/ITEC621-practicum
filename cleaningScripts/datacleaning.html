resolve_project_path <- function(...) {
  candidates <- c(
    normalizePath(file.path("..", ...), winslash = "/", mustWork = FALSE),
    normalizePath(file.path(".", ...), winslash = "/", mustWork = FALSE)
  )

  existing_candidate <- candidates[file.exists(candidates)][1]
  if (!is.na(existing_candidate)) {
    return(existing_candidate)
  }

  candidates[[1]]
}

3 Cleaning Logic

3.1 Helper Functions

join_columns <- function(df, columns, output_column) {
  existing_columns <- columns[columns %in% colnames(df)]
  if (length(existing_columns) == 0) {
    return(df)
  }

  df[[output_column]] <- apply(df[, existing_columns, drop = FALSE], 1, function(row) {
    paste(na.omit(row), collapse = " ")
  })

  df <- df[, !colnames(df) %in% existing_columns]
  df
}

drop_prefixed_columns <- function(df, prefix, max_index) {
  columns_to_drop <- paste0(prefix, "_", 1:max_index)
  columns_to_drop <- columns_to_drop[columns_to_drop %in% colnames(df)]

  if (length(columns_to_drop) > 0) {
    df <- df[, !colnames(df) %in% columns_to_drop]
  }

  df
}

select_500_columns <- function(df, max_index = 27) {
  allowed_columns <- c("linkedinUrl", "jobDescription", "jobTitle", "companyName")
  allowed_columns <- c(
    allowed_columns,
    paste0("jobStartedOn_", 1:max_index),
    paste0("jobEndedOn_", 1:max_index)
  )
  allowed_columns <- allowed_columns[allowed_columns %in% colnames(df)]

  df[, allowed_columns, drop = FALSE]
}

3.2 Experience Cleaning Function

clean_full_experience_df <- function(df, max_index = 27) {
  job_description_columns <- paste0("jobDescription_", 1:max_index)
  title_columns <- paste0("title_", 1:max_index)
  company_name_columns <- paste0("companyName_", 1:max_index)

  df <- join_columns(df, job_description_columns, "jobDescription")
  df <- join_columns(df, title_columns, "jobTitle")
  df <- join_columns(df, company_name_columns, "companyName")

  df <- drop_prefixed_columns(df, "companyId", max_index)
  df <- drop_prefixed_columns(df, "companySize", max_index)
  df <- drop_prefixed_columns(df, "companyIndustry", max_index)
  df <- drop_prefixed_columns(df, "companyWebsite", max_index)
  df <- drop_prefixed_columns(df, "employmentType", max_index)
  df <- drop_prefixed_columns(df, "jobLocation", max_index)
  df <- drop_prefixed_columns(df, "jobLocationCountry", max_index)
  df <- drop_prefixed_columns(df, "jobStillWorking", max_index)

  select_500_columns(df, max_index)
}

3.3 Dataset-Specific Cleaning Functions

clean_500_dataset <- function() {
  df_experiences <- read_csv(data_500_experience_full_path, show_col_types = FALSE)
  df_current_experiences <- read_csv(data_500_experience_current_path, show_col_types = FALSE)
  df_skills <- read_csv(data_500_skills_path, show_col_types = FALSE)

  list(
    experiences = clean_full_experience_df(df_experiences),
    current_experiences = clean_full_experience_df(df_current_experiences, max_index = 1),
    skills = df_skills
  )
}

clean_5k_dataset <- function() {
  df_experiences <- read_excel(data_5k_experience_path, sheet = "Scraped data")
  df_skills <- read_csv(data_5k_skills_path, show_col_types = FALSE)

  columns_to_drop <- c(
    "current_company_url",
    "Company_size",
    "last_scraped_date",
    "previous_job_start_date",
    "previous_job_location",
    "previous_job_country",
    "previous_company_size",
    "employment_type",
    "is_currently_employed",
    "current_job_country",
    "current_job_location",
    "current_job_start_date",
    "email",
    "full_name",
    "education_start_year",
    "education_end_year",
    "previous_job_end_date",
    "total_experience_years",
    "current_job_duration_years"
  )

  columns_to_drop <- columns_to_drop[columns_to_drop %in% colnames(df_experiences)]
  if (length(columns_to_drop) > 0) {
    df_experiences <- df_experiences[, !colnames(df_experiences) %in% columns_to_drop]
  }

  list(
    experiences = df_experiences,
    skills = df_skills
  )
}

3.4 Save Functions

save_cleaned_data <- function(data_list) {
  write_csv(data_list$experiences, file.path(processed_dir, "cleaned_500_experiences.csv"))
  write_csv(
    data_list$current_experiences,
    file.path(processed_dir, "cleaned_500_current_experiences.csv")
  )
  write_csv(data_list$skills, file.path(processed_dir, "cleaned_500_skills.csv"))
}

save_cleaned_5k_data <- function(data_list) {
  write_csv(data_list$experiences, file.path(processed_dir, "cleaned_5k_experiences.csv"))
  write_csv(data_list$skills, file.path(processed_dir, "cleaned_5k_skills.csv"))
}
